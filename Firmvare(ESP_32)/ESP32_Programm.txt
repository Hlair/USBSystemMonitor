// Універсальне видобування метрик (підтримує анімовані та сирі дані)
void get_metric_values(uint8_t base_mode, bool animated, float &v1, float &v2, bool &isDual) {
    v1 = v2 = 0.0f; 
    isDual = false;
    
    switch(base_mode) {
        case 10: v1 = animated ? animCpu.current_val : currentMetrics.cpu_total; break;
        case 20: v1 = animated ? animRam.current_val : currentMetrics.ram_percent; break;
        case 30: 
            v1 = (animated ? animNetRx.current_val : currentMetrics.net_rx_kbps) * 8.0f / 1024.0f;
            v2 = (animated ? animNetTx.current_val : currentMetrics.net_tx_kbps) * 8.0f / 1024.0f;
            isDual = true; 
            break;
        default:
            if (base_mode >= 40 && base_mode <= 49) {
                uint8_t idx = base_mode - 40;
                if (idx < currentMetrics.num_gpus) {
                    v1 = animated ? animGpus[idx].current_val : currentMetrics.gpu_loads[idx];
                }
            } else if (base_mode >= 50 && base_mode <= 59) {
                uint8_t idx = base_mode - 50;
                if (idx < currentMetrics.num_gpus) {
                    v1 = currentMetrics.gpu_vram_percents[idx]; // VRAM не анімується
                }
            } else if (base_mode >= 60 && base_mode <= 69) {
                uint8_t idx = base_mode - 60;
                if (idx < currentMetrics.num_disks) {
                    v1 = (animated ? animDisksRead[idx].current_val : currentMetrics.disk_read_kbps[idx]) / 1024.0f;
                    v2 = (animated ? animDisksWrite[idx].current_val : currentMetrics.disk_write_kbps[idx]) / 1024.0f;
                    isDual = true;
                }
            }
            break;
    }
}

// Формування універсальних заголовків для графіків та екранів
void set_metric_title(lv_obj_t* label, uint8_t base_mode, const char* default_net_title) {
    if (base_mode == 10) lv_label_set_text(label, "CPU Total");
    else if (base_mode == 20) lv_label_set_text(label, "RAM Usage");
    else if (base_mode == 30) lv_label_set_text(label, default_net_title);
    else if (base_mode >= 40 && base_mode <= 49) lv_label_set_text_fmt(label, "GPU %d Load", base_mode - 40);
    else if (base_mode >= 50 && base_mode <= 59) lv_label_set_text_fmt(label, "GPU %d VRAM", base_mode - 50);
    else if (base_mode >= 60 && base_mode <= 69) lv_label_set_text_fmt(label, "Disk %d", base_mode - 60);
}

// Плавне змішування кольорів для індикації навантаження
lv_color_t get_core_color(int percent) {
    lv_color_t c_green = lv_color_hex(HEX_NORD_GPU);
    lv_color_t c_yellow = lv_color_hex(HEX_NORD_RAM);
    lv_color_t c_red = lv_color_hex(HEX_NORD_RED);

    if (percent < 50) return lv_color_mix(c_yellow, c_green, (percent * 255) / 50);
    return lv_color_mix(c_red, c_yellow, ((percent - 50) * 255) / 50);
}

// Розрахунок контрольної суми пакету
uint8_t CalculateChecksum(const uint8_t* data, size_t length) {
    uint8_t checksum = 0;
    for (size_t i = 0; i < length; ++i) checksum ^= data[i];
    return checksum;
}

void my_disp_flush(lv_disp_drv_t *disp_drv, const lv_area_t *area, lv_color_t *color_p) {
    uint32_t w = (area->x2 - area->x1 + 1);
    uint32_t h = (area->y2 - area->y1 + 1);
    tft.drawRGBBitmap(area->x1, area->y1, (uint16_t *)&color_p->full, w, h);
    lv_disp_flush_ready(disp_drv);
}

lv_obj_t* create_metric_arc(lv_obj_t * parent, int radius, int thickness, uint32_t color_hex) {
    lv_obj_t * arc = lv_arc_create(parent);
    lv_obj_set_size(arc, radius * 2, radius * 2);
    lv_obj_center(arc);
    lv_arc_set_rotation(arc, 270);
    lv_arc_set_bg_angles(arc, 0, 360);
    lv_arc_set_range(arc, 0, 100);
    lv_arc_set_value(arc, 0);
    lv_obj_remove_style(arc, NULL, LV_PART_KNOB);
    lv_obj_clear_flag(arc, LV_OBJ_FLAG_CLICKABLE);

    lv_obj_set_style_arc_width(arc, thickness, LV_PART_MAIN);
    lv_obj_set_style_arc_color(arc, lv_color_hex(HEX_NORD_BG_LIGHT), LV_PART_MAIN);
    lv_obj_set_style_arc_width(arc, thickness, LV_PART_INDICATOR);
    lv_obj_set_style_arc_color(arc, lv_color_hex(color_hex), LV_PART_INDICATOR);
    lv_obj_set_style_arc_rounded(arc, true, LV_PART_INDICATOR);
    return arc;
}

    lv_obj_t * rect = lv_obj_create(row);

    lv_obj_set_size(rect, 10, 10);
    lv_obj_set_style_radius(rect, 3, 0);
    lv_obj_set_style_bg_opa(rect, 255, 0);
    lv_obj_set_style_bg_color(rect, lv_color_hex(color_hex), 0);
    lv_obj_align(rect, LV_ALIGN_LEFT_MID, 0, 0);

    lv_obj_set_style_text_color(label, lv_color_hex(HEX_NORD_TEXT), 0);

    lv_label_set_text(label, text);

}

// Налаштування сітки ядер
void configure_cores_grid(uint8_t num_cores) {
    if (num_cores == 0 || num_cores > 32 || configured_cores == num_cores) return;
    
    int cols, rows;
    if (num_cores <= 2) { cols = num_cores; rows = 1; }
    else if (num_cores <= 4) { cols = 2; rows = 2; }
    else if (num_cores <= 6) { cols = 3; rows = 2; }
    else if (num_cores <= 8) { cols = 4; rows = 2; }
    else if (num_cores <= 12) { cols = 4; rows = 3; }
    else if (num_cores <= 16) { cols = 4; rows = 4; }
    else if (num_cores <= 24) { cols = 6; rows = 4; }
    else { cols = 8; rows = 4; }
    
    int gap = 4; 
    int box_w = (240 - gap * (cols + 1)) / cols;
    int box_h = (240 - gap * (rows + 1)) / rows;
    const lv_font_t * font = (box_h >= 60) ? &lv_font_montserrat_24 : &lv_font_montserrat_14; 
    
    for (int i = 0; i < 32; i++) {
        if (i < num_cores) {
            int x = gap + (i % cols) * (box_w + gap);
            int y = gap + (i / cols) * (box_h + gap);
            lv_obj_set_size(core_boxes[i], box_w, box_h);
            lv_obj_align(core_boxes[i], LV_ALIGN_TOP_LEFT, x, y); 
            lv_obj_set_style_text_font(core_labels[i], font, 0);
            lv_obj_clear_flag(core_boxes[i], LV_OBJ_FLAG_HIDDEN);
        } else {
            lv_obj_add_flag(core_boxes[i], LV_OBJ_FLAG_HIDDEN);
        }
    }
    configured_cores = num_cores;
}

// Налаштування макету дашборда
void configure_summary_layout(uint8_t num_gpus, uint8_t num_disks) {
    if (sum_configured_gpus == num_gpus && sum_configured_disks == num_disks) return;
    int total_w = 232;
    int gap = 4;

    if (num_gpus > 0) {
        int gpu_w = (total_w - gap * (num_gpus - 1)) / num_gpus;
        for(int i = 0; i < 4; i++) {
            if (i < num_gpus) {
                lv_obj_set_width(box_sum_gpu[i], gpu_w);
                lv_obj_clear_flag(box_sum_gpu[i], LV_OBJ_FLAG_HIDDEN);
            } else lv_obj_add_flag(box_sum_gpu[i], LV_OBJ_FLAG_HIDDEN);
        }
    }
    if (num_disks > 0) {
        int disk_w = (total_w - gap * (num_disks - 1)) / num_disks;
        for(int i = 0; i < 4; i++) {
            if (i < num_disks) {
                lv_obj_set_width(box_sum_disk[i], disk_w);
                lv_obj_clear_flag(box_sum_disk[i], LV_OBJ_FLAG_HIDDEN);
            } else lv_obj_add_flag(box_sum_disk[i], LV_OBJ_FLAG_HIDDEN);
        }
    }
    sum_configured_gpus = num_gpus;
    sum_configured_disks = num_disks;
}

// =====================================================================
// КОМУНІКАЦІЯ ТА ОБРОБКА ПАКЕТІВ (ЯДРО 0)
// =====================================================================
void processPacket(uint8_t* buffer, uint16_t totalLength) {
    PacketHeader* header = (PacketHeader*)buffer;
    switch (header->command) {
        case HANDSHAKE:
            Serial.write(0xBB);
            xSemaphoreTake(dataMutex, portMAX_DELAY); 
            if (!isConnected) isConnected = true;
            xSemaphoreGive(dataMutex); 
            break;
            
        case METRICS_BASIC:
            if (header->payload_length == sizeof(BasicMetricsPayload)) {
                xSemaphoreTake(dataMutex, portMAX_DELAY);
                memcpy(&currentMetrics, buffer + sizeof(PacketHeader), sizeof(BasicMetricsPayload));
                newDataFlag = true; 
                xSemaphoreGive(dataMutex);
            }
            break;
            
        case SET_DISPLAY_MODE:
            if (header->payload_length == 1) {
                xSemaphoreTake(dataMutex, portMAX_DELAY);
                currentDisplayMode = buffer[sizeof(PacketHeader)];
                xSemaphoreGive(dataMutex);
            }
            break;
    }
}

void UartTaskCode(void *pvParameters) {
    unsigned long lastRxByteTime = 0;
    const unsigned long RX_TIMEOUT_MS = 50;
    
    for(;;) {
        if (rxState != WAIT_MAGIC && (millis() - lastRxByteTime > RX_TIMEOUT_MS)) {
            rxState = WAIT_MAGIC;
            rxIndex = 0;
        }

        while (Serial.available() > 0) {
            uint8_t b = Serial.read();
            lastRxByteTime = millis(); 
            
            if (rxState == WAIT_MAGIC) {
                if (b == 0xAA) { rxBuffer[0] = b; rxIndex = 1; rxState = WAIT_HEADER; }
            } 
            else if (rxState == WAIT_HEADER) {
                if (rxIndex >= sizeof(rxBuffer)) { rxState = WAIT_MAGIC; rxIndex = 0; continue; }
                rxBuffer[rxIndex++] = b;
                if (rxIndex == sizeof(PacketHeader)) {
                    memcpy(&tempHeader, rxBuffer, sizeof(PacketHeader));
                    rxState = (tempHeader.payload_length > 128) ? WAIT_MAGIC : WAIT_PAYLOAD;
                    if (rxState == WAIT_MAGIC) rxIndex = 0;
                }
            } 
            else if (rxState == WAIT_PAYLOAD) {
                if (rxIndex >= sizeof(rxBuffer)) { rxState = WAIT_MAGIC; rxIndex = 0; continue; }
                rxBuffer[rxIndex++] = b;
                uint16_t expectedLength = sizeof(PacketHeader) + tempHeader.payload_length + 1; 
                
                if (rxIndex == expectedLength) {
                    uint8_t receivedChecksum = rxBuffer[rxIndex - 1];
                    uint8_t calculatedChecksum = CalculateChecksum(rxBuffer, rxIndex - 1);
                    if (receivedChecksum == calculatedChecksum) {
                        xSemaphoreTake(dataMutex, portMAX_DELAY);
                        lastDataReceived = millis();
                        xSemaphoreGive(dataMutex);
                        processPacket(rxBuffer, rxIndex);
                    }
                    rxState = WAIT_MAGIC;
                    rxIndex = 0;
                }
            }
        }

        xSemaphoreTake(dataMutex, portMAX_DELAY);
        bool connectedTemp = isConnected;
        unsigned long lastRecvTemp = lastDataReceived;
        xSemaphoreGive(dataMutex);
        
        if (connectedTemp && lastRecvTemp > 0 && (millis() - lastRecvTemp > 3000)) {
            xSemaphoreTake(dataMutex, portMAX_DELAY);
            isConnected = false;
            lastDataReceived = 0;
            xSemaphoreGive(dataMutex);
        }
        vTaskDelay(1 / portTICK_PERIOD_MS);
    }
}

    tft.invertDisplay(true); 
    tft.fillScreen(NORD_BG);

    // Ініціалізація LVGL
    lv_init();
    lv_disp_draw_buf_init(&draw_buf, buf1, NULL, SCREEN_WIDTH * SCREEN_HEIGHT / 10);
    static lv_disp_drv_t disp_drv;
    lv_disp_drv_init(&disp_drv);
    disp_drv.hor_res = SCREEN_WIDTH;
    disp_drv.ver_res = SCREEN_HEIGHT;
    disp_drv.flush_cb = my_disp_flush;
    disp_drv.draw_buf = &draw_buf;
    lv_disp_drv_register(&disp_drv);

    // Побудова екранів
    build_splash_screen();
    build_metric_screen();
    build_chart_screen();
    build_cores_screen();
    build_net_radar_screen();
    build_general_screen();
    build_big_num_screen();     
    build_speedometer_screen(); 
    build_summary_screen();
    
    lv_scr_load(scr_splash);

    dataMutex = xSemaphoreCreateMutex();
    xTaskCreatePinnedToCore(UartTaskCode, "UART_Task", 10000, NULL, 1, NULL, 0);
}

void loop() {
    // 1. Оновлення таймерів LVGL
    static uint32_t last_tick = millis();
    uint32_t current_tick = millis();
    lv_tick_inc(current_tick - last_tick);
    last_tick = current_tick;
    lv_timer_handler();

    xSemaphoreTake(dataMutex, portMAX_DELAY);
    bool localConnected = isConnected;
    BasicMetricsPayload renderMetrics = currentMetrics;
    uint8_t localMode = currentDisplayMode;
    bool localNewData = newDataFlag;
    
    // 2. Ініціалізація анімації при надходженні нових даних
    if (newDataFlag) {
        animCpu.old_val = animCpu.current_val; animCpu.target_val = renderMetrics.cpu_total;
        animRam.old_val = animRam.current_val; animRam.target_val = renderMetrics.ram_percent;
        animNetRx.old_val = animNetRx.current_val; animNetRx.target_val = renderMetrics.net_rx_kbps;
        animNetTx.old_val = animNetTx.current_val; animNetTx.target_val = renderMetrics.net_tx_kbps;

        for(int i = 0; i < renderMetrics.num_gpus && i < 4; i++) {
            animGpus[i].old_val = animGpus[i].current_val;
            animGpus[i].target_val = renderMetrics.gpu_loads[i];
        }
        for(int i = 0; i < renderMetrics.num_disks && i < 4; i++) {
            animDisksRead[i].old_val = animDisksRead[i].current_val;
            animDisksRead[i].target_val = renderMetrics.disk_read_kbps[i];
            animDisksWrite[i].old_val = animDisksWrite[i].current_val;
            animDisksWrite[i].target_val = renderMetrics.disk_write_kbps[i];
        }
        for(int c = 0; c < renderMetrics.num_cores && c < 32; c++) {
            animCores[c].old_val = animCores[c].current_val;
            animCores[c].target_val = renderMetrics.core_loads[c];
        }

        animStartTime = millis();
        newDataFlag = false;
    }
    xSemaphoreGive(dataMutex);
    
    // 3. Логіка відображення
    if (localConnected) {
        // Керування режимами
        if (!metricUIShown || activeDisplayMode != localMode) {
            if (localMode == 0) { lv_scr_load(scr_metrics); } 
            else if (localMode == 1) { lv_scr_load(scr_net_radar); } 
            else if (localMode == 2) { configure_cores_grid(renderMetrics.num_cores); lv_scr_load(scr_cores); } 
            else if (localMode == 3) { lv_scr_load(scr_general); }
            else if (localMode == 5) { configure_summary_layout(renderMetrics.num_gpus, renderMetrics.num_disks); lv_scr_load(scr_summary); }
            else if (localMode >= 10 && localMode <= 69) { configure_chart_for_metric(localMode); lv_scr_load(scr_chart); }
            else if (localMode >= 70 && localMode <= 129) { configure_big_num_for_metric(localMode); lv_scr_load(scr_big_num); }
            else if (localMode >= 130 && localMode <= 189) { configure_speedometer_for_metric(localMode); lv_scr_load(scr_speedometer); }
            else { lv_scr_load(scr_metrics); }
            
            metricUIShown = true;
            activeDisplayMode = localMode;
        }

        // Оновлення динамічних графіків
        if (localNewData) {
            if (activeDisplayMode == 1) {
                float v1, v2; bool isDual;
                get_metric_values(30, false, v1, v2, isDual); // Використовуємо сирі дані для графіку
                
                int rx_val = min(32000, (int)v1);
                int tx_val = min(32000, (int)v2);
                lv_chart_set_next_value(chart_radar, ser_rx, rx_val);
                lv_chart_set_next_value(chart_radar, ser_tx, -tx_val);

                lv_label_set_text_fmt(label_rx_speed, "DL: %d.%d Mbps", SPLIT_F(v1));
                lv_label_set_text_fmt(label_tx_speed, "UL: %d.%d Mbps", SPLIT_F(v2));

                int maxVisibleVal = 0;
                lv_coord_t * y_arr_rx = lv_chart_get_y_array(chart_radar, ser_rx);
                lv_coord_t * y_arr_tx = lv_chart_get_y_array(chart_radar, ser_tx);
                uint16_t point_cnt = lv_chart_get_point_count(chart_radar);
                for (uint16_t i = 0; i < point_cnt; i++) {
                    if (y_arr_rx[i] != LV_CHART_POINT_NONE && y_arr_rx[i] > maxVisibleVal) maxVisibleVal = y_arr_rx[i];
                    if (y_arr_tx[i] != LV_CHART_POINT_NONE && -y_arr_tx[i] > maxVisibleVal) maxVisibleVal = -y_arr_tx[i];
                }
                int newTargetMaxY = max(150, maxVisibleVal + (maxVisibleVal / 5));
                if (currentRadarMaxY != newTargetMaxY) {
                    currentRadarMaxY = newTargetMaxY;
                    lv_chart_set_range(chart_radar, LV_CHART_AXIS_PRIMARY_Y, -currentRadarMaxY, currentRadarMaxY);
                }
            }
            else if (activeDisplayMode >= 10 && activeDisplayMode <= 69) {
                float v1, v2; bool isDual;
                get_metric_values(activeDisplayMode, false, v1, v2, isDual);
                
                int val1 = min(32000, (int)v1);
                int val2 = min(32000, (int)v2);
                lv_chart_set_next_value(chart_metric, ser_metric_1, val1);
                if (isDual) lv_chart_set_next_value(chart_metric, ser_metric_2, val2);

                if (activeDisplayMode == 30 || (activeDisplayMode >= 60 && activeDisplayMode <= 69)) {
                    int maxVisibleVal = 0;
                    int minCeiling = (activeDisplayMode == 30) ? 100 : 200;
                    
                    lv_coord_t * y_arr_1 = lv_chart_get_y_array(chart_metric, ser_metric_1);
                    lv_coord_t * y_arr_2 = lv_chart_get_y_array(chart_metric, ser_metric_2);
                    uint16_t point_cnt = lv_chart_get_point_count(chart_metric);
                    for (uint16_t i = 0; i < point_cnt; i++) {
                        if (y_arr_1[i] != LV_CHART_POINT_NONE && y_arr_1[i] > maxVisibleVal) maxVisibleVal = y_arr_1[i];
                        if (isDual && y_arr_2[i] != LV_CHART_POINT_NONE && y_arr_2[i] > maxVisibleVal) maxVisibleVal = y_arr_2[i];
                    }
                    int newTargetMaxY = max(minCeiling, maxVisibleVal + (maxVisibleVal / 5));
                    if (currentChartMaxY != newTargetMaxY) {
                        currentChartMaxY = newTargetMaxY;
                        lv_chart_set_range(chart_metric, LV_CHART_AXIS_PRIMARY_Y, 0, currentChartMaxY);
                    }
                }
            }
        }

        // 4. Інтерполяція даних анімації
        unsigned long elapsed = millis() - animStartTime;
        float progress = min(1.0f, (float)elapsed / ANIM_DURATION);
        
        animCpu.current_val = animCpu.old_val + (animCpu.target_val - animCpu.old_val) * progress;
        animRam.current_val = animRam.old_val + (animRam.target_val - animRam.old_val) * progress;
        animNetRx.current_val = animNetRx.old_val + (animNetRx.target_val - animNetRx.old_val) * progress;
        animNetTx.current_val = animNetTx.old_val + (animNetTx.target_val - animNetTx.old_val) * progress;
        
        for(int i = 0; i < 4; i++) {
            animGpus[i].current_val = animGpus[i].old_val + (animGpus[i].target_val - animGpus[i].old_val) * progress;
            animDisksRead[i].current_val = animDisksRead[i].old_val + (animDisksRead[i].target_val - animDisksRead[i].old_val) * progress;
            animDisksWrite[i].current_val = animDisksWrite[i].old_val + (animDisksWrite[i].target_val - animDisksWrite[i].old_val) * progress;
        }
        for(int c = 0; c < renderMetrics.num_cores && c < 32; c++) {
            animCores[c].current_val = animCores[c].old_val + (animCores[c].target_val - animCores[c].old_val) * progress;
        }

        // 5. Оновлення віджетів на екрані
        update_lvgl_widgets();
    } 
    else {
        // Обробка втрати з'єднання
        if (metricUIShown) {
            lv_scr_load(scr_splash);
            metricUIShown = false;
            activeDisplayMode = -1; 
        }
    }

    delay(5);
}
